    import { DotMap } from '../../Types/DotMap';

/**
 * DotMap Google Maps
 */
export class DotMapGoogleMaps extends DotMap {

    /**
     * Creates an instance of DotMapGoogleMaps.
     * @param {BGMapWrapper} bgmap - Background map object.
     * @param {JSON} geometry - JSON object with the geometry that was read from the file.
     * @param {Object} userOptions - User options to be used.
     * @memberOf Choropleth
     */
    constructor(bgmap, geometry, userOptions) {
        super(bgmap, geometry, userOptions);
    }

    /**
     * M8) Setup all events used by the API. Right now the API uses: drag, zoom and click events.
     * Pan = drag, Zoom = zoom_changed, click = click
     * @param {number} mappos - This map id. 
     * @return {void}
     * @see https://developers.google.com/maps/documentation/javascript/events
     */
    setupEvents(mappos) {
        /*if(!this.gisplayOptions.interactive && this.gisplayOptions.mapOnClickFunction === undefined)
            return;*/
        this.bGMap.addEventListener('zoom_changed', () => { this.draw(); });
        this.bGMap.addEventListener('center_changed', () => { this.draw(); });
        this.bGMap.addEventListener('drag', () => { this.draw(); });
        this.bGMap.addEventListener('click', e => {
            const lat = e.latLng.lat();
            const lon = e.latLng.lng();
            if (this.rtree !== undefined)
                this.searchRTree(lon, lat);
            if (this.kdtree !== undefined)
                this.searchKdTree(lon, lat);
        });
    }
}
